# Client-Side Dependencies Installation Commands
# Run these commands from the client directory

# Install all dependencies at once (recommended)
npm install

# OR install dependencies individually:

# ---------- Production Dependencies ----------

# Stripe payment integration
npm install @stripe/react-stripe-js@^5.4.1
npm install @stripe/stripe-js@^8.5.3

# Tailwind CSS PostCSS plugin
npm install @tailwindcss/postcss@^4.1.17

# HTTP client for API requests
npm install axios@^1.13.2

# Firebase for authentication
npm install firebase@^12.6.0

# Animation library
npm install framer-motion@^12.23.25

# PDF generation
npm install jspdf@^3.0.4
npm install jspdf-autotable@^5.0.2

# React core libraries
npm install react@^19.2.0
npm install react-dom@^19.2.0

# Form handling
npm install react-hook-form@^7.68.0

# Icon library
npm install react-icons@^5.5.0

# Routing
npm install react-router-dom@^7.10.1

# Charts and data visualization
npm install recharts@^3.5.1

# Beautiful alerts and modals
npm install sweetalert2@^11.26.3


# ---------- Development Dependencies ----------

# ESLint and plugins for code quality
npm install --save-dev @eslint/js@^9.39.1
npm install --save-dev eslint@^9.39.1
npm install --save-dev eslint-plugin-react-hooks@^7.0.1
npm install --save-dev eslint-plugin-react-refresh@^0.4.24
npm install --save-dev globals@^16.5.0

# TypeScript type definitions
npm install --save-dev @types/react@^19.2.5
npm install --save-dev @types/react-dom@^19.2.3

# Vite build tool and React plugin
npm install --save-dev @vitejs/plugin-react@^5.1.1
npm install --save-dev vite@^7.2.4

# Tailwind CSS and PostCSS
npm install --save-dev tailwindcss@^4.1.17
npm install --save-dev postcss@^8.5.6
npm install --save-dev autoprefixer@^10.4.22


# ---------- Quick Setup Commands ----------

# 1. Navigate to client directory
cd client

# 2. Install all dependencies
npm install

# 3. Create .env file with required environment variables
# VITE_FIREBASE_API_KEY=your_firebase_api_key
# VITE_FIREBASE_AUTH_DOMAIN=your_firebase_auth_domain
# VITE_FIREBASE_PROJECT_ID=your_firebase_project_id
# VITE_FIREBASE_STORAGE_BUCKET=your_firebase_storage_bucket
# VITE_FIREBASE_MESSAGING_SENDER_ID=your_firebase_messaging_sender_id
# VITE_FIREBASE_APP_ID=your_firebase_app_id
# VITE_API_URL=http://localhost:5000/api
# VITE_STRIPE_PUBLISHABLE_KEY=your_stripe_publishable_key

# 4. Run development server
npm run dev
